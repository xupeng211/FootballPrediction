  ```yaml
    - name: api_alerts
      rules:
        - alert: HighErrorRate
          expr: rate(http_requests_total{status=~"5.."}[5m]) > 0.05
          for: 5m
          labels:
            severity: critical
          annotations:
            summary: "API错误率过高"
            description: "错误率 {{ $value }}% 超过阈值5%"

        - alert: SlowResponse
          expr: histogram_quantile(0.95, rate(http_request_duration_seconds_bucket[5m])) > 1
          for: 10m
          labels:
            severity: warning
          annotations:
            summary: "API响应缓慢"
  ```
