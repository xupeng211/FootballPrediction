### 改进 3: 内联模糊匹配

**之前**:
```python
normalizer = TeamNameNormalizer()
score = normalizer.fuzzy_match(db_home, m['home_team'])
```

**之后**:
```python
from thefuzz import fuzz

def simple_normalize(name: str) -> str:
    return name.lower().strip()

score = fuzz.token_sort_ratio(
    simple_normalize(db_home),
    simple_normalize(m['home_team'])
)
```

### 改进 4: 内联数据库连接

**之前**:
```python
settings = get_settings()
conn = psycopg2.connect(
    host=settings.database.host,
    port=settings.database.port,
    ...
)
```

**之后**:
```python
def get_db_connection():
    host = os.environ.get('DB_HOST', 'localhost')
    port = int(os.environ.get('DB_PORT', '5432'))
    database = os.environ.get('DB_NAME', 'football_db')
    user = os.environ.get('DB_USER', 'football_user')
    password = os.environ.get('DB_PASSWORD', 'football_password')

    return psycopg2.connect(
        host=host, port=port, database=database,
        user=user, password=password
    )
```

## 技术架构

### Playwright API 拦截（V134.1 优化版）

```python
async def capture_page_134(url: str, page_num: int):
    captured_data = None
    capture_lock = False  # 防止重复读取

    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        context = await browser.new_context()
        page = await context.new_page()

        async def handle_response(response):
            nonlocal captured_data, capture_lock

            # 只捕获目标 API，且未锁定时
            if ('ajax-sport-country-tournament-archive' in response.url and
                not capture_lock and captured_data is None):

                try:
                    body = await response.body()

                    # 记录所有 API 调用
                    logger.debug(f"📡 API 调用: {len(body):,} 字节")

                    if len(body) > MIN_PAYLOAD_SIZE:
                        captured_data = body
                        capture_lock = True
                        logger.info(f"📦 捕获: {len(body):,} 字节")

                except Exception as e:
                    # 防御性错误处理
                    pass

        page.on('response', handle_response)

        # 使用 domcontentloaded 等待策略
        await page.goto(url, timeout=90000, wait_until="domcontentloaded")
        await asyncio.sleep(3)

        # 简化滚动（10 次，500px）
        for i in range(10):
            await page.mouse.wheel(0, 500)
            await asyncio.sleep(0.2)

        await asyncio.sleep(3)

    return captured_data
```
