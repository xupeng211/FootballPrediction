### 方案 B: 创建新的数据复活脚本

**新脚本**: `scripts/ops/v41_103_resurrect_xg.py`

```python
import json
import requests
from src.config_unified import get_settings

FOTMOB_BASE_URL = "https://www.fotmob.com/api"

def extract_xg_from_match_details(match_id: str) -> dict | None:
    """从 matchDetails 端点提取 xG 数据"""
    url = f"{FOTMOB_BASE_URL}/matchDetails?matchId={match_id}"
    headers = {
        "User-Agent": "Mozilla/5.0 (Windows NT 10.0; Win64; x64) ...",
        "Accept": "application/json, text/plain, */*",
        "Referer": "https://www.fotmob.com/",
    }

    response = requests.get(url, headers=headers, timeout=30)

    if response.status_code != 200:
        return None

    data = response.json()

    # 提取 xG 数据
    try:
        stats = data["content"]["stats"]["Periods"]["All"]["stats"]
        for stat in stats:
            if stat["title"] == "Expected goals (xG)":
                return {
                    "home_xg": float(stat["stats"][0]),
                    "away_xg": float(stat["stats"][1]),
                }
    except (KeyError, IndexError):
        pass

    return None
```
