```sql
-- 检查 technical_features 字段填充率
SELECT
    COUNT(*) as total_matches,
    COUNT(technical_features) as with_features,
    COUNT(technical_features) * 100.0 / COUNT(*) as fill_rate
FROM matches;

-- 检查 xG 覆盖率
SELECT
    COUNT(*) as total_matches,
    COUNT(technical_features->'home_xg') as with_xg,
    COUNT(technical_features->'home_xg') * 100.0 / COUNT(*) as xg_coverage
FROM matches;

-- 按赛季统计
SELECT
    season,
    COUNT(*) as total_matches,
    COUNT(technical_features->'home_xg') as with_xg,
    ROUND(COUNT(technical_features->'home_xg') * 100.0 / COUNT(*), 2) as xg_coverage
FROM matches
GROUP BY season
ORDER BY season;
```
