```python
def update_match_hash(
    conn,
    match_id: str,
    hash_value: str,
    url: str,
    dry_run: bool = False
) -> bool:
    """V41.33 增强：处理 UNIQUE 约束冲突"""

    try:
        # 尝试插入哈希
        cur.execute(upsert_query, (...))
        conn.commit()
        return True

    except psycopg2.IntegrityError as e:
        # V41.33: 检测到 UNIQUE 约束冲突
        if "duplicate key" in str(e).lower() and "unique constraint" in str(e).lower():
            # 查找冲突的现有记录
            cur.execute("""
                SELECT fotmob_id, home_team, away_team, mapping_method
                FROM matches_mapping
                WHERE oddsportal_hash = %s AND fotmob_id != %s
            """, (hash_value, match_id))

            # 更新 review_status
            cur.execute("""
                UPDATE matches_mapping
                SET review_status = 'hash_conflict_review',
                    review_notes = %s
                WHERE fotmob_id = %s
            """, (str(conflict_info), match_id))

            conn.commit()
            return False  # 未成功更新哈希，但系统继续运行
```

**特性**:
- 自动检测 UNIQUE 约束冲突
- 标记为 `hash_conflict_review`
- 记录冲突详细信息（existing_fotmob_id, existing_match, existing_method）
- 不中断采集流程
- 返回 False 表示未成功更新哈希

```python
def find_matching_hash(matches, target, normalizer):
    # 1. 精确匹配
    if match['home_team'] == target.home_team:
        return match

    # 2. 模糊匹配
    if normalizer.are_same_team(match['home_team'], target.home_team):
        return match

    # 3. 词根匹配
    home_root = re.sub(r'[\s\-]+', '', target.home_team).lower()
    db_home_root = re.sub(r'[\s\-]+', '', match['home_team']).lower()
    if home_root == db_home_root:
        return match
```
