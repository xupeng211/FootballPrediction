```python
def validate_cross_date(
    self,
    target_date: datetime,
    actual_date: datetime,
    tolerance_hours: int = 24
) -> bool:
    """验证日期是否在容差范围内"""
    if actual_date is None:
        return False
    delta = abs(target_date - actual_date)
    return delta <= timedelta(hours=tolerance_hours)
```

```python
#!/usr/bin/env python3
"""V41.48: 工业级系统健康检查脚本"""

import sys
import logging
from pathlib import Path

# 添加项目根目录到路径
sys.path.insert(0, str(Path(__file__).parent.parent))

from src.config_unified import get_settings
import psycopg2
import yaml

# 配置日志
logging.basicConfig(level=logging.INFO, format='[%(levelname)s] %(message)s')
logger = logging.getLogger(__name__)

# 健康检查结果
results = {
    "pass": 0,
    "warn": 0,
    "fail": 0,
    "details": []
}


def check(name: str, status: str, message: str):
    """记录检查结果"""
    results[status] += 1
    status_icon = {"pass": "✅", "warn": "⚠️", "fail": "❌"}[status]
    results["details"].append(f"{status_icon} {name}: {message}")
    logger.info(f"{status_icon} {name}: {message}")


def main():
    logger.info("🚀 V41.48 系统健康检查启动")
    logger.info("=" * 60)

    # 1. 数据库连接检查
    try:
        settings = get_settings()
        conn = psycopg2.connect(
            host=settings.database.host,
            port=settings.database.port,
            database=settings.database.name,
            user=settings.database.user,
            password=settings.database.password.get_secret_value(),
        )
        conn.close()
        check("数据库连接", "pass", f"{settings.database.host}:{settings.database.port}/{settings.database.name}")
    except Exception as e:
        check("数据库连接", "fail", str(e))

    # 2. 数据库连接池配置检查
    if settings.database.pool_size >= 10:
        check("连接池大小", "pass", f"pool_size={settings.database.pool_size}")
    else:
        check("连接池大小", "warn", f"pool_size={settings.database.pool_size} (建议 >= 10)")

    # 3. YAML 配置完整性检查
    try:
        yaml_path = Path("config/leagues.yaml")
        if yaml_path.exists():
            with open(yaml_path, "r") as f:
                config = yaml.safe_load(f)

            leagues = config.get("leagues", {})
            total = len(leagues)
            with_slug = sum(1 for l in leagues.values() if l.get("oddsportal_slug"))

            if total == with_slug:
                check("YAML 配置", "pass", f"{total} 个联赛全部配置 oddsportal_slug")
            else:
                check("YAML 配置", "warn", f"{with_slug}/{total} 联赛配置 oddsportal_slug")
        else:
            check("YAML 配置", "fail", "config/leagues.yaml 不存在")
    except Exception as e:
        check("YAML 配置", "fail", str(e))

    # 4. 代理端口检查
    import socket
    healthy_ports = 0
    for port in range(7890, 7900):
        try:
            sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
            sock.settimeout(1)
            result = sock.connect_ex(('127.0.0.1', port))
            sock.close()
            if result == 0:
                healthy_ports += 1
        except:
            pass

    if healthy_ports >= 5:
        check("代理端口", "pass", f"{healthy_ports}/10 个端口可用")
    elif healthy_ports >= 2:
        check("代理端口", "warn", f"{healthy_ports}/10 个端口可用 (建议 >= 5)")
    else:
        check("代理端口", "fail", f"{healthy_ports}/10 个端口可用 (要求 >= 2)")

    # 5. MatchInfo 模型检查
    try:
        from src.services.hash_alignment_service import MatchInfo
        # 测试实例化
        test_match = MatchInfo(
            home_team="Test Home",
            away_team="Test Away",
            hash_value="12345678",
            url="test"
        )
        check("MatchInfo 模型", "pass", "模型定义正确")
    except Exception as e:
        check("MatchInfo 模型", "fail", str(e))

    # 6. 动态 URL 引擎检查
    try:
        from src.services.hash_alignment_service import HashAlignmentService

        # 测试跨年赛季
        format_cross = HashAlignmentService.detect_season_format("23/24")
        format_calendar = HashAlignmentService.detect_season_format("2024")

        if format_cross == "cross_year" and format_calendar == "calendar_year":
            check("赛季格式检测", "pass", "跨年 & 自然年格式正确识别")
        else:
            check("赛季格式检测", "warn", "格式识别异常")
    except Exception as e:
        check("赛季格式检测", "fail", str(e))

    # 打印汇总
    logger.info("")
    logger.info("=" * 60)
    logger.info("📊 健康检查汇总:")
    logger.info(f"   通过: {results['pass']}")
    logger.info(f"   警告: {results['warn']}")
    logger.info(f"   失败: {results['fail']}")
    logger.info("=" * 60)

    # 打印详细信息
    for detail in results["details"]:
        logger.info(detail)

    # 判断是否满足"一键启动全球收割"条件
    logger.info("")
    if results['fail'] == 0 and results['pass'] >= 5:
        logger.info("🎉 系统满足'一键启动全球收割'条件！")
        logger.info("   您可以安全运行:")
        logger.info("   python main.py --action align-hashes --run-harvest")
        return 0
    else:
        logger.error("❌ 系统不满足'一键启动全球收割'条件，请先解决上述问题")
        return 1


if __name__ == "__main__":
    sys.exit(main())
```
