def save_raw_payload(league: str, season: str, page: int, decrypted_data: str) -> Optional[str]:
    """V135.1: 保存原始 JSON 数据（生矿持久化）."""
    os.makedirs(RAW_PAYLOADS_DIR, exist_ok=True)

    safe_league = re.sub(r'[^\w\s-]', '', league).strip().replace(' ', '_')
    safe_season = season.replace('/', '-')

    filename = f"{safe_league}_{safe_season}_p{page}.json"
    filepath = os.path.join(RAW_PAYLOADS_DIR, filename)

    with open(filepath, 'w', encoding='utf-8') as f:
        f.write(decrypted_data)

    return filepath

```python
def get_pending_tasks_from_queue(limit: Optional[int] = None) -> List[Tuple[str, str, int]]:
    """V135.1: 从 match_search_queue 提取待处理任务."""
    # V135.1: 优先级排序 - 五大联赛优先
    cur.execute(f"""
        WITH league_priority AS (
            SELECT
                m.league_name,
                m.season,
                COUNT(*) as missing_count,
                COALESCE(p.priority, 100) as priority_value
            FROM matches m
            LEFT JOIN (VALUES
                ('Premier League', 1),
                ('La Liga', 2),
                ('Serie A', 3),
                ('Bundesliga', 4),
                ('Ligue 1', 5)
            ) p(league_name, priority) ON m.league_name = p.league_name
            WHERE m.oddsportal_url IS NULL
            GROUP BY m.league_name, m.season, p.priority
            ORDER BY priority_value ASC, missing_count DESC
        )
        SELECT league_name, season, missing_count
        FROM league_priority
    """)
```
