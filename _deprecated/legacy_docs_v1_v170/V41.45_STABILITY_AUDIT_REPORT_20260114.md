```python
# V41.45: 端口健康检查
def check_proxy_port_health(self, port: int, timeout: float = 1.0) -> bool:
    """
    检查代理端口健康状态
    - 通过 socket.connect_ex() 检测端口可用性
    - 超时 1 秒快速检测
    - 异常捕获，返回 False
    """
    try:
        sock = socket.socket(socket.AF_INET, socket.SOCK_STREAM)
        sock.settimeout(timeout)
        result = sock.connect_ex(('127.0.0.1', port))
        sock.close()
        return result == 0
    except (socket.error, OSError):
        return False

# V41.45: 获取健康端口（带排除列表）
def get_healthy_proxy_port(
    self,
    excluded_ports: set[int] | None = None,
    max_attempts: int = None
) -> int | None:
    """
    获取健康的代理端口（带容错机制）
    - 自动排除已知失效端口
    - 打乱顺序避免总是尝试相同端口
    - 达到 max_attempts 后返回 None
    """

# V41.45: 设置代理端口（带重试）
def set_proxy_port_with_retry(
    self,
    port: int | None = None,
    excluded_ports: set[int] | None = None,
    max_retry: int = 3
) -> int:
    """
    设置代理端口（带重试和容错）
    - 检查首选端口是否可用
    - 失败时尝试其他端口
    - 所有端口失效时使用降级策略
    """
```

**核心逻辑**:
```
1. 检查首选端口 → 可用？返回 : 尝试其他
2. 排除失效端口 → 避免重复尝试
3. 健康检查 → socket.connect_ex()
4. 降级策略 → 所有失败时仍返回端口
```
