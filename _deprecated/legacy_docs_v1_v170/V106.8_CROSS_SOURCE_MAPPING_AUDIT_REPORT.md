```python
async def extract_pinnacle_odds(page: Page) -> dict | None:
    """提取 Pinnacle 初盘和终盘赔率"""

    # 等待数据加载
    await page.wait_for_selector('.odds-table', timeout=10000)

    # 查找 Pinnacle 行
    # 选择器: 需要根据实际页面结构调整
    pinnacle_row = await page.query_selector('[data-bookmaker="Pinnacle"]')

    if not pinnacle_row:
        return None

    # 提取初盘赔率 (opening odds)
    opening_home = await pinnacle_row.query_selector('.opening-home')
    opening_draw = await pinnacle_row.query_selector('.opening-draw')
    opening_away = await pinnacle_row.query_selector('.opening-away')

    # 提取终盘赔率 (closing odds)
    closing_home = await pinnacle_row.query_selector('.closing-home')
    closing_draw = await pinnacle_row.query_selector('.closing-draw')
    closing_away = await pinnacle_row.query_selector('.closing-away')

    return {
        'opening': {
            'home': float(await opening_home.inner_text()) if opening_home else None,
            'draw': float(await opening_draw.inner_text()) if opening_draw else None,
            'away': float(await opening_away.inner_text()) if opening_away else None,
        },
        'closing': {
            'home': float(await closing_home.inner_text()) if closing_home else None,
            'draw': float(await closing_draw.inner_text()) if closing_draw else None,
            'away': float(await closing_away.inner_text()) if closing_away else None,
        },
        'extracted_at': datetime.now().isoformat()
    }
```

#### 3.2 数据完整性验证

```python
def validate_odds_integrity(odds: dict) -> bool:
    """验证赔率数据的完整性"""

    if not odds:
        return False

    # 检查初盘赔率
    opening = odds.get('opening', {})
    if not all([opening.get('home'), opening.get('draw'), opening.get('away')]):
        return False

    # 检查终盘赔率
    closing = odds.get('closing', {})
    if not all([closing.get('home'), closing.get('draw'), closing.get('away')]):
        return False

    # 完整性评分: Score = 1/P1 + 1/P2 + 1/P3
    # 有效范围: 1.02 < Score < 1.08
    score = (
        1 / opening['home'] +
        1 / opening['draw'] +
        1 / opening['away']
    )

    if not (1.02 < score < 1.08):
        return False

    return True
```

```python
    matches: List[dict],
    max_concurrent: int = 5
) -> List[dict]:
    """限制并发数的收割逻辑"""

    semaphore = asyncio.Semaphore(max_concurrent)
```
