```python
# 代理端口列表 (10 个隔离端口)
PROXY_PORTS = list(range(7890, 7900))  # 7890-7899

@dataclass
class HarvestConfig:
    """收割配置"""
    proxy_port: int  # 每个 Worker 绑定一个端口
    limit: int
    delay_min: float
    delay_max: float
    batch_id: str


def worker_process(config: HarvestConfig) -> WorkerResult:
    """
    工作进程 - 独立进程，绑定单个代理端口

    进程间代理隔离:
    - Worker 1: 7890
    - Worker 2: 7891
    - Worker 3: 7892
    ...
    - Worker 10: 7899
    """
    # 设置代理环境变量
    os.environ['PROXY_PORT'] = str(config.proxy_port)

    # 数据库 SKIP LOCKED 保证并发安全
    FOR UPDATE SKIP LOCKED
```

```sql
SELECT
    m.league_name,
    m.season,
    COUNT(DISTINCT m.match_id) as total_matches,
    COUNT(DISTINCT mm.fotmob_id) as with_hash,
    ROUND(
        100.0 * COUNT(DISTINCT mm.fotmob_id) / COUNT(DISTINCT m.match_id),
        2
    ) as coverage_pct
FROM matches m
LEFT JOIN matches_mapping mm ON m.match_id = mm.fotmob_id
WHERE m.season = '23/24'
  AND m.league_name IN ('Premier League', 'La Liga', 'Bundesliga', 'Serie A', 'Ligue 1')

ORDER BY coverage_pct DESC;
```
