## Step 4: 重新处理 801 条 FAILED 记录

**重新处理脚本** (Step 4 会提供):

```python
# scripts/maintenance/reprocess_failed_matches.py
import sys
sys.path.insert(0, '/home/user/projects/FootballPrediction')

from src.config import get_settings
from src.processors.v25_production_extractor import V25ProductionExtractor
import psycopg2
from psycopg2.extras import RealDictCursor

def reprocess_failed_matches(limit: int = 100):
    """
    重新处理 FAILED 状态的记录

    策略:
    1. 查询 l2_raw_json 不为空的 FAILED 记录
    2. 使用 V25ProductionExtractor 重新提取特征
    3. 更新 l2_extracted_features 和状态
    """
    settings = get_settings()

    conn = psycopg2.connect(
        host=settings.database.host,
        database=settings.database.name,
        user=settings.database.user,
        password=settings.database.password.get_secret_value(),
        cursor_factory=RealDictCursor,
    )

    # 1. 获取需要重新处理的记录
    with conn.cursor() as cur:
        cur.execute("""
            SELECT match_id, l2_raw_json, home_team, away_team
            FROM matches
            WHERE collection_status = 'FAILED'
              AND l2_raw_json IS NOT NULL
            ORDER BY extracted_at DESC
            LIMIT %s
        """, (limit,))

        failed_matches = cur.fetchall()

    print(f"找到 {len(failed_matches)} 条可重新处理的记录")

    extractor = V25ProductionExtractor()
    success_count = 0

    for match in failed_matches:
        try:
            # 2. 重新提取特征
            result = extractor.extract(match['l2_raw_json'])

            # 3. 保存结果
            with conn.cursor() as cur:
                cur.execute("""
                    UPDATE matches SET
                        l2_extracted_features = %s::jsonb,
                        l2_data_version = %s,
                        extracted_at = NOW(),
                        collection_status = %s,
                        last_error = NULL,
                        updated_at = NOW()
                    WHERE match_id = %s
                """, (
                    json.dumps(result.features),
                    extractor.version,
                    result.status.value,
                    match['match_id']
                ))

            conn.commit()
            success_count += 1
            print(f"✅ {match['match_id']}: {match['home_team']} vs {match['away_team']}")

        except Exception as e:
            print(f"❌ {match['match_id']}: {e}")
            conn.rollback()

    print(f"\n重新处理完成: {success_count}/{len(failed_matches)} 成功")
    return success_count

if __name__ == "__main__":
    reprocess_failed_matches()
```
